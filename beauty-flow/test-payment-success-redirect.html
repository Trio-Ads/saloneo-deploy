Débogage

Si vous rencontrez des problèmes avec les redirections, vous pouvez utiliser le script de débogage suivant dans la console du navigateur après avoir été redirigé :

// Exécuter ce script dans la console après redirection
(function() {
  console.log('=== DÉBOGAGE REDIRECTION PAIEMENT ===');
  
  // Afficher l'URL complète
  console.log('URL complète:', window.location.href);
  
  // Analyser les paramètres d'URL
  const params = new URLSearchParams(window.location.search);
  console.log('Paramètres d\'URL:', Object.fromEntries(params.entries()));
  
  // Vérifier les paramètres spécifiques
  console.log('- success:', params.get('success'));
  console.log('- error:', params.get('error'));
  console.log('- plan:', params.get('plan'));
  console.log('- transaction_id:', params.get('transaction_id'));
  
  return 'Débogage terminé';
})();
